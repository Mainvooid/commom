333  bool parse(const std::string &arg) noexcept
334  {
335  std::vector<std::string> args;
336  std::string buf;
337  bool in_quote = false;//是否有""
338  for (std::string::size_type i = 0; i < arg.length(); i++) {
339  if (arg[i] == '\"') {
340  in_quote = !in_quote;
341  continue;
342  }
343  if (arg[i] == ' ' && !in_quote) {
344  args.push_back(buf);
345  buf = "";
346  continue;
347  }
348  if (arg[i] == '\\') { //跳过'\'
349  i++;
350  if (i >= arg.length()) {
351  errors.push_back("unexpected occurrence of '\\' at end of string");
352  return false;
353  }
354  }
355  buf += arg[i];
356  }
357  if (in_quote) {
358  errors.push_back("quote is not closed");
359  return false;
360  }
361  if (buf.length() > 0) {
362  args.push_back(buf);
363  }
364  for (size_t i = 0; i < args.size(); i++) {
365  std::cout << "\"" << args[i] << "\"" << std::endl;
366  }
367  return parse(args);
368  }

391  bool parse(int argc, const char * const argv[]) noexcept
392  {
393  errors.clear();
394  others.clear();
395  if (argc < 1) {
396  errors.push_back("argument number must be bigger than 0");
397  return false;
398  }
399  if (prog_name == "") { prog_name = argv[0]; }
400 
401  std::map<char, std::string> lookup;
402  for (std::map<std::string, option_base*>::iterator p = options.begin(); p != options.end(); p++) {
403  if (p->first.length() == 0) {
404  continue;//key不存在
405  }
406  char initial = p->second->short_name();
407  if (initial) {
408  if (lookup.count(initial) > 0) {
409  lookup[initial] = "";
410  errors.push_back(std::string("short option '") + initial + "' is ambiguous");
411  return false;
412  }
413  else {
414  lookup[initial] = p->first;
415  }
416  }
417  }
418 
419  for (int i = 1; i < argc; i++) {
420  if (strncmp(argv[i], "--", 2) == 0) {//长名称
421  const char *p = strchr(argv[i] + 2, '=');
422  if (p) {
423  std::string name(argv[i] + 2, p);
424  std::string val(p + 1);
425  set_option(name, val);
426  }
427  else {
428  std::string name(argv[i] + 2);
429  if (options.count(name) == 0) {
430  errors.push_back("undefined option: --" + name);
431  continue;
432  }
433  if (options[name]->has_value()) {
434  if (i + 1 >= argc) {
435  errors.push_back("option needs value: --" + name);
436  continue;
437  }
438  else {
439  i++;
440  set_option(name, argv[i]);
441  }
442  }
443  else {
444  set_option(name); //bool
445  }
446  }
447  }
448  else if (strncmp(argv[i], "-", 1) == 0) {//短名称
449  if (!argv[i][1]) {//若'-'后无值
450  continue;
451  }
452  char last = argv[i][1];
453  for (int j = 2; argv[i][j]; j++) {
454  last = argv[i][j];
455  if (lookup.count(argv[i][j - 1]) == 0) {
456  errors.push_back(std::string("undefined short option: -") + argv[i][j - 1]);
457  continue;
458  }
459  if (lookup[argv[i][j - 1]] == "") {
460  errors.push_back(std::string("ambiguous short option: -") + argv[i][j - 1]);
461  continue;
462  }
463  set_option(lookup[argv[i][j - 1]]);
464  }
465 
466  if (lookup.count(last) == 0) {
467  errors.push_back(std::string("undefined short option: -") + last);
468  continue;
469  }
470  if (lookup[last] == "") {
471  errors.push_back(std::string("ambiguous short option: -") + last);
472  continue;
473  }
474 
475  if (i + 1 < argc && options[lookup[last]]->has_value()) {
476  set_option(lookup[last], argv[i + 1]);
477  i++;
478  }
479  else {
480  set_option(lookup[last]);
481  }
482  }
483  else {
484  others.push_back(argv[i]);
485  }
486  }
487 
488  for (std::map<std::string, option_base*>::iterator p = options.begin(); p != options.end(); p++) {
489  if (!p->second->valid()) {
490  errors.push_back("need option: --" + std::string(p->first));
491  }
492  }
493  return errors.size() == 0;
494  }
