334  bool parse(const std::string &arg) noexcept
335  {
336  std::vector<std::string> args;
337  std::string buf;
338  bool in_quote = false;//是否有""
339  for (std::string::size_type i = 0; i < arg.length(); i++) {
340  if (arg[i] == '\"') {
341  in_quote = !in_quote;
342  continue;
343  }
344  if (arg[i] == ' ' && !in_quote) {
345  args.push_back(buf);
346  buf = "";
347  continue;
348  }
349  if (arg[i] == '\\') { //跳过'\'
350  i++;
351  if (i >= arg.length()) {
352  errors.push_back("unexpected occurrence of '\\' at end of string");
353  return false;
354  }
355  }
356  buf += arg[i];
357  }
358  if (in_quote) {
359  errors.push_back("quote is not closed");
360  return false;
361  }
362  if (buf.length() > 0) {
363  args.push_back(buf);
364  }
365  for (size_t i = 0; i < args.size(); i++) {
366  std::cout << "\"" << args[i] << "\"" << std::endl;
367  }
368  return parse(args);
369  }

392  bool parse(int argc, const char * const argv[]) noexcept
393  {
394  errors.clear();
395  others.clear();
396  if (argc < 1) {
397  errors.push_back("argument number must be bigger than 0");
398  return false;
399  }
400  if (prog_name == "") { prog_name = argv[0]; }
401 
402  std::map<char, std::string> lookup;
403  for (std::map<std::string, option_base*>::iterator p = options.begin(); p != options.end(); p++) {
404  if (p->first.length() == 0) {
405  continue;//key不存在
406  }
407  char initial = p->second->short_name();
408  if (initial) {
409  if (lookup.count(initial) > 0) {
410  lookup[initial] = "";
411  errors.push_back(std::string("short option '") + initial + "' is ambiguous");
412  return false;
413  }
414  else {
415  lookup[initial] = p->first;
416  }
417  }
418  }
419 
420  for (int i = 1; i < argc; i++) {
421  if (strncmp(argv[i], "--", 2) == 0) {//长名称
422  const char *p = strchr(argv[i] + 2, '=');
423  if (p) {
424  std::string name(argv[i] + 2, p);
425  std::string val(p + 1);
426  set_option(name, val);
427  }
428  else {
429  std::string name(argv[i] + 2);
430  if (options.count(name) == 0) {
431  errors.push_back("undefined option: --" + name);
432  continue;
433  }
434  if (options[name]->has_value()) {
435  if (i + 1 >= argc) {
436  errors.push_back("option needs value: --" + name);
437  continue;
438  }
439  else {
440  i++;
441  set_option(name, argv[i]);
442  }
443  }
444  else {
445  set_option(name); //bool
446  }
447  }
448  }
449  else if (strncmp(argv[i], "-", 1) == 0) {//短名称
450  if (!argv[i][1]) {//若'-'后无值
451  continue;
452  }
453  char last = argv[i][1];
454  for (int j = 2; argv[i][j]; j++) {
455  last = argv[i][j];
456  if (lookup.count(argv[i][j - 1]) == 0) {
457  errors.push_back(std::string("undefined short option: -") + argv[i][j - 1]);
458  continue;
459  }
460  if (lookup[argv[i][j - 1]] == "") {
461  errors.push_back(std::string("ambiguous short option: -") + argv[i][j - 1]);
462  continue;
463  }
464  set_option(lookup[argv[i][j - 1]]);
465  }
466 
467  if (lookup.count(last) == 0) {
468  errors.push_back(std::string("undefined short option: -") + last);
469  continue;
470  }
471  if (lookup[last] == "") {
472  errors.push_back(std::string("ambiguous short option: -") + last);
473  continue;
474  }
475 
476  if (i + 1 < argc && options[lookup[last]]->has_value()) {
477  set_option(lookup[last], argv[i + 1]);
478  i++;
479  }
480  else {
481  set_option(lookup[last]);
482  }
483  }
484  else {
485  others.push_back(argv[i]);
486  }
487  }
488 
489  for (std::map<std::string, option_base*>::iterator p = options.begin(); p != options.end(); p++) {
490  if (!p->second->valid()) {
491  errors.push_back("need option: --" + std::string(p->first));
492  }
493  }
494  return errors.size() == 0;
495  }
