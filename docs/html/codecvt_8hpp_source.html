1 /*
2 @brief Character encoding conversion
3 @author guobao.v@gmail.com
4 @note Unicode(宽字符集)和ANSI(多字节字符集)是编码标准
5  ANSI以单字节(8bit)存放英文字符，以双字节(16bit)存放中文等字符
6  Unicode下，英文和中文的字符都以双字节存放(16bit)
7 
8  以下3个是Unicode标准的实现方式:
9  UTF-16用2个字节来编码所有的字符(标准Unicode)
10  UTF-32则选择用4个字节来编码
11  UTF-8为了网络传输节省资源,采用变长编码(1-6字节编码)
12 
13  std::string 可以是ANSI编码(char 8bit,中文16bit(gb2312))
14  也可以是UTF-8编码(窄多字节编码，中文24bit，需转为ANSI或UNICODE后才能正常显示)
15  std::wstring 可以是标准Unicode编码(wchar_t 16bit)
16  也可以是UTF-8编码(宽多字节编码)
17  wchar_t 在windows的MSVC环境下为UTF-16编码
18  在linux的GCC环境下为UTF-32编码
19  char16_t UTF-16编码
20  char32_t UTF-32编码
21 */

61  inline static std::string utf16_to_utf8(const std::u16string& utf16_string) noexcept
62  {
63  std::string result = u8"";
64 #if _MSC_VER >= 1900
65  std::wstring_convert<std::codecvt_utf8_utf16<int16_t>, int16_t> cvt;
66  auto p = reinterpret_cast<const int16_t*>(utf16_string.data());
67  try {
68  result = cvt.to_bytes(p, p + utf16_string.size());
69  }
70 #else
71  std::wstring_convert<std::codecvt_utf8_utf16<char16_t>, char16_t> cvt;
72  try {
73  result = cvt.to_bytes(utf16_string);
74  }
75 #endif
76  catch (const std::range_error&) {
77  return u8"";
78  }
79  return result;
80  }
