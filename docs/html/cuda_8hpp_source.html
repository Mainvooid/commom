58  template <typename T>
59  void checkCudaRet(T result, char const *const func, const char *const file, int const line) {
60  if (result) {
61 #if defined(_UNICODE) or defined(UNICODE)
62  std::wostringstream oss;
63 #else
64  std::ostringstream oss;
65 #endif
66  oss << cudaGetErrorName(result);
67  common::LOGE(oss.str(), func, file, line);
68  cudaDeviceReset();
69  // Make sure we call CUDA Device Reset before exiting
70  exit(EXIT_FAILURE);
71  }
72  }

98  template<typename R, typename ...FArgs, typename ...Args>
99  float getCudaFnDuration(std::function<R(FArgs...)> Fn, Args&... args) {
100  float duration;
101  cudaEvent_t start, stop;
102  checkCudaRet(cudaEventCreate(&start));
103  checkCudaRet(cudaEventCreate(&stop));
104  checkCudaRet(cudaEventRecord(start, 0));
105  Fn(args...);
106  checkCudaRet(cudaEventRecord(stop, 0));
107  checkCudaRet(cudaEventSynchronize(stop));
108  checkCudaRet(cudaEventElapsedTime(&duration, start, stop));
109  checkCudaRet(cudaEventDestroy(start));
110  checkCudaRet(cudaEventDestroy(stop));
111  return duration;
112  }

162  static void getD3D11Adapter(IDXGIAdapter** pAdapter) noexcept(false)
163  {
164  Microsoft::WRL::ComPtr<IDXGIFactory> pFactory;
165  HRESULT hr = CreateDXGIFactory(__uuidof(IDXGIFactory), (void**)(pFactory.GetAddressOf()));
166  if (FAILED(hr)) {
167  throw std::runtime_error("No DXGI Factory created");
168  }
169 
170  Microsoft::WRL::ComPtr<IDXGIAdapter> _pAdapter;
171  for (UINT i = 0; *pAdapter == nullptr; ++i)
172  {
173  //获取一个候选DXGI适配器
174  hr = pFactory->EnumAdapters(i, _pAdapter.GetAddressOf());
175  if (FAILED(hr)) { break; }
176 
177  //查询是否存在相应的计算设备
178  int cuDevice;
179  checkCudaRet(cudaD3D11GetDevice(&cuDevice, _pAdapter.Get()));
180  Release_s(*pAdapter);
181  *pAdapter = _pAdapter.Get();
182  (*pAdapter)->AddRef();
183  }
184  if (*pAdapter == nullptr) {
185  throw std::runtime_error("No IDXGIAdapter");
186  }
187  }

215  cudaError_t texture2d_to_gpumat(ID3D11Texture2D* p_src_texture, cv::cuda::GpuMat& dst_gpumat)
216  {
217  HRESULT hr = common::windows::texture2d_to_texture2d(p_src_texture, mt_texture_2d.p_d3d11_texture_2d.GetAddressOf(), mp_d3d11_device.Get());
218  if (FAILED(hr)) { return cudaError::cudaErrorUnknown; }
219  D3D11_TEXTURE2D_DESC desc;
220  p_src_texture->GetDesc(&desc);
221 
222  //注册Direct3D 11资源pD3DResource以供CUDA访问
223  checkCudaRet(cudaGraphicsD3D11RegisterResource(&mt_texture_2d.cuda_resource, mt_texture_2d.p_d3d11_texture_2d.Get(), cudaGraphicsRegisterFlagsNone));
224 
225  //映射图形资源以供CUDA访问
226  checkCudaRet(cudaGraphicsMapResources(1, &mt_texture_2d.cuda_resource));
227 
228  //获取一个数组，通过该数组访问映射图形资源的子资源
229  checkCudaRet(cudaGraphicsSubResourceGetMappedArray(&mt_texture_2d.cuda_array, mt_texture_2d.cuda_resource, 0, 0));
230 
231  checkCudaRet(cudaBindTextureToArray(mt_texture_2d.texture_ref, mt_texture_2d.cuda_array, &mt_texture_2d.cuda_array_desc));
232 
233  dst_gpumat.create(desc.Height, desc.Width, CV_MAKETYPE(CV_8U, sizeof(uchar4)));
234 
235  //src和dst的step可能不同
236  checkCudaRet(cudaMemcpy2DFromArray(dst_gpumat.data, dst_gpumat.step,
237  mt_texture_2d.cuda_array, 0, 0, dst_gpumat.cols * sizeof(uchar4), dst_gpumat.rows, cudaMemcpyDeviceToDevice));
238 
239  cudaUnbindTexture(mt_texture_2d.texture_ref);
240 
241  checkCudaRet(cudaGraphicsUnmapResources(1, &mt_texture_2d.cuda_resource));
242  checkCudaRet(cudaGraphicsUnregisterResource(mt_texture_2d.cuda_resource));
243  return cudaError::cudaSuccess;
244  };
