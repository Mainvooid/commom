55  void checkCudaRet(T result, char const *const func, const char *const file, int const line) {
56  if (result) {
57 #if defined(_UNICODE) or defined(UNICODE)
58  std::wostringstream oss;
59 #else
60  std::ostringstream oss;
61 #endif
62  oss << cudaGetErrorName(result);
63  common::LOGE(oss.str(), func, file, line);
64  cudaDeviceReset();
65  // Make sure we call CUDA Device Reset before exiting
66  exit(EXIT_FAILURE);
67  }
68  }

90  template<typename R, typename ...FArgs, typename ...Args>
91  static float getCudaFnDuration(std::function<R(FArgs...)> Fn, Args&... args) {
92  float duration;
93  cudaEvent_t start, stop;
94  checkCudaRet(cudaEventCreate(&start));
95  checkCudaRet(cudaEventCreate(&stop));
96  checkCudaRet(cudaEventRecord(start, 0));
97  Fn(args...);
98  checkCudaRet(cudaEventRecord(stop, 0));
99  checkCudaRet(cudaEventSynchronize(stop));
100  checkCudaRet(cudaEventElapsedTime(&duration, start, stop));
101  checkCudaRet(cudaEventDestroy(start));
102  checkCudaRet(cudaEventDestroy(stop));
103  return duration;
104  }

154  static void getD3D11Adapter(IDXGIAdapter** pAdapter) noexcept(false)
155  {
156  Microsoft::WRL::ComPtr<IDXGIFactory> pFactory;
157  HRESULT hr = CreateDXGIFactory(__uuidof(IDXGIFactory), (void**)(pFactory.GetAddressOf()));
158  if (FAILED(hr)) {
159  throw std::runtime_error("No DXGI Factory created");
160  }
161 
162  Microsoft::WRL::ComPtr<IDXGIAdapter> _pAdapter;
163  for (UINT i = 0; *pAdapter == nullptr; ++i)
164  {
165  //获取一个候选DXGI适配器
166  hr = pFactory->EnumAdapters(i, _pAdapter.GetAddressOf());
167  if (FAILED(hr)) { break; }
168 
169  //查询是否存在相应的计算设备
170  int cuDevice;
171  checkCudaRet(cudaD3D11GetDevice(&cuDevice, _pAdapter.Get()));
172  Release_s(*pAdapter);
173  *pAdapter = _pAdapter.Get();
174  (*pAdapter)->AddRef();
175  }
176  if (*pAdapter == nullptr) {
177  throw std::runtime_error("No IDXGIAdapter");
178  }
179  }

207  cudaError_t texture2d_to_gpumat(ID3D11Texture2D* p_src_texture, cv::cuda::GpuMat& dst_gpumat)
208  {
209  HRESULT hr = common::windows::texture2d_to_texture2d(p_src_texture, mt_texture_2d.p_d3d11_texture_2d.GetAddressOf(), mp_d3d11_device.Get());
210  if (FAILED(hr)) { return cudaError::cudaErrorUnknown; }
211  D3D11_TEXTURE2D_DESC desc;
212  p_src_texture->GetDesc(&desc);
213 
214  //注册Direct3D 11资源pD3DResource以供CUDA访问
215  checkCudaRet(cudaGraphicsD3D11RegisterResource(&mt_texture_2d.cuda_resource, mt_texture_2d.p_d3d11_texture_2d.Get(), cudaGraphicsRegisterFlagsNone));
216 
217  //映射图形资源以供CUDA访问
218  checkCudaRet(cudaGraphicsMapResources(1, &mt_texture_2d.cuda_resource));
219 
220  //获取一个数组，通过该数组访问映射图形资源的子资源
221  checkCudaRet(cudaGraphicsSubResourceGetMappedArray(&mt_texture_2d.cuda_array, mt_texture_2d.cuda_resource, 0, 0));
222 
223  checkCudaRet(cudaBindTextureToArray(mt_texture_2d.texture_ref, mt_texture_2d.cuda_array, &mt_texture_2d.cuda_array_desc));
224 
225  dst_gpumat.create(desc.Height, desc.Width, CV_MAKETYPE(CV_8U, sizeof(uchar4)));
226 
227  //src和dst的step可能不同
228  checkCudaRet(cudaMemcpy2DFromArray(dst_gpumat.data, dst_gpumat.step,
229  mt_texture_2d.cuda_array, 0, 0, dst_gpumat.cols * sizeof(uchar4), dst_gpumat.rows, cudaMemcpyDeviceToDevice));
230 
231  cudaUnbindTexture(mt_texture_2d.texture_ref);
232 
233  checkCudaRet(cudaGraphicsUnmapResources(1, &mt_texture_2d.cuda_resource));
234  checkCudaRet(cudaGraphicsUnregisterResource(mt_texture_2d.cuda_resource));
235  return cudaError::cudaSuccess;
236  };
