64  static void checkCudaRet(cudaError_t result, char const *const func, const char *const file, int const line)
65  {
66  if (result) {
67  std::ostringstream oss;
68  oss << cudaGetErrorName(result);
69  common::LOGE(oss.str(), func, file, line);
70 
71  cudaDeviceReset();
72  // Make sure we call CUDA Device Reset before exiting
73  exit(EXIT_FAILURE);
74  }
75  }

101  template<typename R, typename ...FArgs, typename ...Args>
102  float getCudaFnDuration(std::function<R(FArgs...)> Fn, Args&... args) {
103  float duration;
104  cudaEvent_t start, stop;
105  checkCudaRet(cudaEventCreate(&start));
106  checkCudaRet(cudaEventCreate(&stop));
107  checkCudaRet(cudaEventRecord(start, 0));
108  Fn(args...);
109  checkCudaRet(cudaEventRecord(stop, 0));
110  checkCudaRet(cudaEventSynchronize(stop));
111  checkCudaRet(cudaEventElapsedTime(&duration, start, stop));
112  checkCudaRet(cudaEventDestroy(start));
113  checkCudaRet(cudaEventDestroy(stop));
114  return duration;
115  }

168  static void getD3D11Adapter(IDXGIAdapter** pAdapter) noexcept(false)
169  {
170  ComPtr<IDXGIFactory> pFactory;
171  HRESULT hr = CreateDXGIFactory(__uuidof(IDXGIFactory), (void**)(pFactory.GetAddressOf()));
172  if (FAILED(hr)) {
173  throw std::runtime_error("No DXGI Factory created");
174  }
175 
176  ComPtr<IDXGIAdapter> _pAdapter;
177  for (UINT i = 0; *pAdapter == nullptr; ++i)
178  {
179  //获取一个候选DXGI适配器
180  hr = pFactory->EnumAdapters(i, _pAdapter.GetAddressOf());
181  if (FAILED(hr)) { break; }
182 
183  //查询是否存在相应的计算设备
184  int cuDevice;
185  checkCudaRet(cudaD3D11GetDevice(&cuDevice, _pAdapter.Get()));
186  Release_s(*pAdapter);
187  *pAdapter = _pAdapter.Get();
188  (*pAdapter)->AddRef();
189  }
190  if (*pAdapter == nullptr) {
191  throw std::runtime_error("No IDXGIAdapter");
192  }
193  }

241  bool texture2d_to_gpumat(ID3D11Texture2D* p_src_texture, cv::cuda::GpuMat& dst_gpumat)
242  {
243  HRESULT hr = common::windows::texture2d_to_texture2d(p_src_texture, mt_texture2d_to_gpumat.p_d3d11_texture_2d.ReleaseAndGetAddressOf(), mp_d3d11_device.Get());
244  if (FAILED(hr)) { return false; }
245  D3D11_TEXTURE2D_DESC desc;
246  p_src_texture->GetDesc(&desc);
247  dst_gpumat.release();
248  dst_gpumat.create(desc.Height, desc.Width, CV_MAKETYPE(CV_8U, sizeof(uchar4)));
249  if (FAILED(hr)) { return false; }
250 
251  //注册Direct3D 11资源以供CUDA访问
252  checkCudaRet(cudaGraphicsD3D11RegisterResource(&mt_texture2d_to_gpumat.cuda_resource, mt_texture2d_to_gpumat.p_d3d11_texture_2d.Get(), cudaGraphicsRegisterFlagsNone));
253  //映射图形资源以供CUDA访问
254  checkCudaRet(cudaGraphicsMapResources(1, &mt_texture2d_to_gpumat.cuda_resource));
255 
256  //获取一个数组，通过该数组访问映射图形资源的子资源
257  checkCudaRet(cudaGraphicsSubResourceGetMappedArray(&mt_texture2d_to_gpumat.cuda_array, mt_texture2d_to_gpumat.cuda_resource, 0, 0));
258 
259  checkCudaRet(cudaBindTextureToArray(mt_texture2d_to_gpumat.texture_ref, mt_texture2d_to_gpumat.cuda_array, &mt_texture2d_to_gpumat.cuda_array_desc));
260  //src和dst的step可能不同
261  checkCudaRet(cudaMemcpy2DFromArray(dst_gpumat.data, dst_gpumat.step,
262  mt_texture2d_to_gpumat.cuda_array, 0, 0, dst_gpumat.cols * sizeof(uchar4), dst_gpumat.rows, cudaMemcpyDeviceToDevice));
263  checkCudaRet(cudaUnbindTexture(mt_texture2d_to_gpumat.texture_ref));
264 
265  checkCudaRet(cudaGraphicsUnmapResources(1, &mt_texture2d_to_gpumat.cuda_resource));
266  checkCudaRet(cudaGraphicsUnregisterResource(mt_texture2d_to_gpumat.cuda_resource));
267  return true;
268  };
