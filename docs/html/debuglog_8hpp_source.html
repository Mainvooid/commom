97  template<typename T>
98  void Log(T msg, const level_e nLevel) const
99  {
100  if (nLevel < m_level || nLevel == level_e::Off) { return; }
101  T _name = tvalue<T>(m_name, codecvt::ansi_to_unicode(m_name));
102  if (m_name == "") { _name = tvalue<T>("G", L"G"); }
103 
104  switch (nLevel)
105  {
106  case level_e::Trace:
107  OutputDebugStringEx()(tvalue<T>("Trace [%s] : %s\n", L"Trace [%s] : %s\n"), _name.data(), msg.data());
108  break;
109  case level_e::Debug:
110  OutputDebugStringEx()(tvalue<T>("Debug [%s] : %s\n", L"Debug [%s] : %s\n"), _name.data(), msg.data());
111  break;
112  case level_e::Info:
113  OutputDebugStringEx()(tvalue<T>("Info [%s] : %s\n", L"Info [%s] : %s\n"), _name.data(), msg.data());
114  break;
115  case level_e::Warn:
116  OutputDebugStringEx()(tvalue<T>("Warn [%s] : %s\n", L"Warn [%s] : %s\n"), _name.data(), msg.data());
117  break;
118  case level_e::Error:
119  OutputDebugStringEx()(tvalue<T>("Error [%s] : %s\n", L"Error [%s] : %s\n"), _name.data(), msg.data());
120  break;
121  case level_e::Fatal:
122  OutputDebugStringEx()(tvalue<T>("Fatal [%s] : %s\n", L"Fatal [%s] : %s\n"), _name.data(), msg.data());
123  break;
124  default:
125  break;
126  }
127  }

135  /*
136  *@brief 打印Debug信息
137  *@param _func 宏 __FUNCTION__ 函数名
138  *@param _file 宏 __FILE__ 文件名
139  *@param _line 宏 __LINE__ 行数
140  */
141  template<typename T>
142  void Log(T msg, const level_e nLevel, const char* _func, const char* _file, const int _line)const
143  {
144  if (nLevel < m_level || nLevel == level_e::Off) { return; }
145  T func = tvalue<T>(_func, codecvt::ansi_to_unicode(_func));
146  T file = tvalue<T>(_file, codecvt::ansi_to_unicode(_file));
147 
148  size_t n = file.find_last_of('\\');
149  n = file.find_last_of('\\', n - 1);
150  n = file.find_last_of('\\', n - 1);
151  if (n == static_cast<size_t>(-1)) { n = 0; }
152  T file_fmt = T(file.begin() + n, file.end());
153 
154  T _name = tvalue<T>(m_name, codecvt::ansi_to_unicode(m_name));
155  if (m_name == "") { _name = tvalue<T>(" ", L" "); }
156 
157  switch (nLevel)
158  {
159  case level_e::Trace:
160  OutputDebugStringEx()(tvalue<T>("Trace [%s] : %s [...%s(%d) %s]\n", L"Trace [%s] : %s [...%s(%d) %s]\n"),
161  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
162  break;
163  case level_e::Debug:
164  OutputDebugStringEx()(tvalue<T>("Debug [%s] : %s [...%s(%d) %s]\n", L"Debug [%s] : %s [...%s(%d) %s]\n"),
165  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
166  break;
167  case level_e::Info:
168  OutputDebugStringEx()(tvalue<T>("Info [%s] : %s [...%s(%d) %s]\n", L"Info [%s] : %s [...%s(%d) %s]\n"),
169  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
170  break;
171  case level_e::Warn:
172  OutputDebugStringEx()(tvalue<T>("Warn [%s] : %s [...%s(%d) %s]\n", L"Warn [%s] : %s [...%s(%d) %s]\n"),
173  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
174  break;
175  case level_e::Error:
176  OutputDebugStringEx()(tvalue<T>("Error [%s] : %s [...%s(%d) %s]\n", L"Error [%s] : %s [...%s(%d) %s]\n"),
177  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
178  break;
179  case level_e::Fatal:
180  OutputDebugStringEx()(tvalue<T>("Fatal [%s] : %s [...%s(%d) %s]\n", L"Fatal [%s] : %s [...%s(%d) %s]\n"),
181  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
182  break;
183  default:
184  break;
185  }
186  }
