49  static void OutputDebugStringEx(const char* format, ...) noexcept
50  {
51  char buf[BUFSIZ];
52  va_list args;//可变长参数列表
53  va_start(args, format);//获取列表第一个参数
54  vsprintf_s(buf, format, args);//按格式执行拼接
55  va_end(args);//清空列表

98  /*
99  *@brief 打印Debug信息
100  */
101  void Log(T msg, const level_e nLevel) const
102  {
103  if (nLevel < m_level || nLevel == level_e::Off) { return; }
104  T _name = m_name;
105  if (m_name == _T("")) { _name = _T("G"); }
106 
107  switch (nLevel)
108  {
109  case level_e::Trace:
110  OutputDebugStringEx(_T("Trace [%s] : %s\n"), _name.data(), msg.data());
111  break;
112  case level_e::Debug:
113  OutputDebugStringEx(_T("Debug [%s] : %s\n"), _name.data(), msg.data());
114  break;
115  case level_e::Info:
116  OutputDebugStringEx(_T("Info [%s] : %s\n"), _name.data(), msg.data());
117  break;
118  case level_e::Warn:
119  OutputDebugStringEx(_T("Warn [%s] : %s\n"), _name.data(), msg.data());
120  break;
121  case level_e::Error:
122  OutputDebugStringEx(_T("Error [%s] : %s\n"), _name.data(), msg.data());
123  break;
124  case level_e::Fatal:
125  OutputDebugStringEx(_T("Fatal [%s] : %s\n"), _name.data(), msg.data());
126  break;
127  default:
128  break;
129  }
130  }

132  /*
133  *@brief 打印Debug信息
134  *@param _func 宏 __FUNCTION__ 函数名
135  *@param _file 宏 __FILE__ 文件名
136  *@param _line 宏 __LINE__ 行数
137  */
138  void Log(T msg, const level_e nLevel, const char* _func, const char* _file, const int _line)const
139  {
140  if (nLevel < m_level || nLevel == level_e::Off) { return; }
141  T func, file;
142 #if defined(_UNICODE) or defined(UNICODE)
143  if (std::is_same<T, std::wstring>::value) {
144  func = codecvt::ansi_to_unicode(_func);
145  file = codecvt::ansi_to_unicode(_file);
146  }
147 #else
148  if (std::is_same<T, std::string>::value) {
149  func = _func;
150  file = _file;
151  }
152 #endif
153  size_t n = file.find_last_of('\\');
154  n = file.find_last_of('\\', n - 1);
155  n = file.find_last_of('\\', n - 1);
156  if (n == static_cast<size_t>(-1)) { n = 0; }
157  T file_fmt = T(file.begin() + n, file.end());
158 
159  T _name = m_name;
160  if (m_name == _T("")) { _name = _T(" "); }
161 
162  switch (nLevel)
163  {
164  case level_e::Trace:
165  OutputDebugStringEx(_T("Trace [%s] : %s [...%s(%d) %s]\n"),
166  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
167  break;
168  case level_e::Debug:
169  OutputDebugStringEx(_T("Debug [%s] : %s [...%s(%d) %s]\n"),
170  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
171  break;
172  case level_e::Info:
173  OutputDebugStringEx(_T("Info [%s] : %s [...%s(%d) %s]\n"),
174  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
175  break;
176  case level_e::Warn:
177  OutputDebugStringEx(_T("Warn [%s] : %s [...%s(%d) %s]\n"),
178  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
179  break;
180  case level_e::Error:
181  OutputDebugStringEx(_T("Error [%s] : %s [...%s(%d) %s]\n"),
182  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
183  break;
184  case level_e::Fatal:
185  OutputDebugStringEx(_T("Fatal [%s] : %s [...%s(%d) %s]\n"),
186  _name.data(), msg.data(), file_fmt.data(), _line, func.data());
187  break;
188  default:
189  break;
190  }
191  }
