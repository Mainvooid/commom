91  static cv::Mat mergeImage(const cv::InputArray& left, const cv::InputArray& right)
92  noexcept(noexcept(!left.empty() && !right.empty()))

94  if (left.empty() || right.empty()) {

100  cv::Mat dst_leftMat = img_merge(cv::Rect(0, 0, left.cols(), left.rows()));
