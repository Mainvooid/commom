124  static cv::Mat mergeImage(cv::InputArray left, cv::InputArray right)
125  noexcept(noexcept(!left.empty() && !right.empty()))
126  {
127  if (left.empty() || right.empty()) {
128  std::ostringstream msg;
129  msg << _TAG << "..InputArray is empty";
130  throw std::runtime_error(msg.str());
131  }
132  cv::Mat _left = left.getMat(), _right = right.getMat();
133  if (left.channels() != right.channels()) {
134  if (left.channels() == 1) {
135  cv::cvtColor(left, _left, cv::COLOR_GRAY2BGRA);
136  }
137  if (left.channels() == 3) {
138  cv::cvtColor(left, _left, cv::COLOR_BGR2BGRA);
139  }
140  if (right.channels() == 1) {
141  cv::cvtColor(right, _right, cv::COLOR_GRAY2BGRA);
142  }
143  if (right.channels() == 3) {
144  cv::cvtColor(right, _right, cv::COLOR_BGR2BGRA);
145  }
146  }
147  cv::Mat img_merge(cv::Size(left.cols() + right.cols(), left.rows()), CV_MAKETYPE(_left.depth(), _left.channels()), cv::Scalar::all(0));
148  cv::Mat dst_leftMat = img_merge(cv::Rect(0, 0, left.cols(), left.rows()));
149  cv::Mat dst_rightMat = img_merge(cv::Rect(left.cols(), 0, right.cols(), right.rows()));
150  _left.copyTo(dst_leftMat);
151  _right.copyTo(dst_rightMat);
152  return img_merge;
153  }

168  void init(cv::InputArrayOfArrays chessboards, cv::Size board_size, size_t square_size) noexcept(false)
169  {
170  chessboards.getMatVector(m_chessboards);
171  m_board_size = board_size;
172  m_square_size = square_size;
173 
174  std::vector<cv::Point3f> temp_obj_point;
175  for (int i = 0; i < m_board_size.height; i++) {
176  for (int j = 0; j < m_board_size.width; j++) {
177  //假设定标板放在世界坐标系中z=0的平面上
178  temp_obj_point.push_back(cv::Point3f(j * m_square_size, i * m_square_size, 0));
179  }
180  }
181  for (size_t i = 0; i < m_chessboards.size(); i++) {
182  cv::Mat chessboard_gray;
183  std::vector<cv::Point2f> corners;//提取的角点坐标
184  //提取角点
185  cv::findChessboardCorners(m_chessboards[i], board_size, corners,
186  cv::CALIB_CB_ADAPTIVE_THRESH + cv::CALIB_CB_NORMALIZE_IMAGE + cv::CALIB_CB_FAST_CHECK);
187  cv::cvtColor(m_chessboards[i], chessboard_gray, cv::COLOR_BGR2GRAY);
188  //亚像素精确化
189  cv::cornerSubPix(chessboard_gray, corners, board_size,
190  cv::Size(-1, -1), cv::TermCriteria(3, 100, std::numeric_limits<float>::epsilon()));
191  m_corners_set.push_back(corners);
192  m_object_points_set.push_back(temp_obj_point);
193  }
194  //摄像机定标
195  int flags = 0;
196  flags |= cv::fisheye::CALIB_RECOMPUTE_EXTRINSIC;//外部优化将在每次迭代后重新计算
197  flags |= cv::fisheye::CALIB_CHECK_COND; //检查条件数的有效性
198  flags |= cv::fisheye::CALIB_FIX_SKEW; //偏斜系数alpha设置为零并保持为零,代表求解时假设内参中fx=fy
199 
200  cv::fisheye::calibrate(m_object_points_set, m_corners_set, m_chessboards[0].size(),
201  m_intrinsic_param_mat, m_distortion_coeffs, m_rotation_vectors, m_translation_vectors, flags,
202  cv::TermCriteria(3, 100, std::numeric_limits<float>::epsilon()));
203 
204  if ((m_distortion_coeffs[0] <= std::numeric_limits<float>::epsilon()
205  && m_distortion_coeffs[1] <= std::numeric_limits<float>::epsilon()
206  && m_distortion_coeffs[2] <= std::numeric_limits<float>::epsilon()
207  && m_distortion_coeffs[3] <= std::numeric_limits<float>::epsilon()
208  ) || (m_intrinsic_param_mat.rows == 0 || m_intrinsic_param_mat.cols == 0)) {
209  throw std::invalid_argument("Invalid output parameters");//可能存在某些标定图无法成功计算出参数,应排除之
210  }
211 #if defined(HAVE_CUDA) && defined(HAVE_CUDA_KERNEL)
212  m_intrinsic_param_mat_g.upload(m_intrinsic_param_mat);
213  m_distortion_coeffs_g.upload(m_distortion_coeffs);
214 #endif // HAVE_CUDA && HAVE_CUDA_KERNEL
215  }

241  void remap(cv::InputArray src, cv::OutputArray dst, double fx = 1.0, double fy = 1.0, double cx = 0, double cy = 0)
242  {
243  //调节视场大小, 乘的系数越小视场越大
244  m_new_intrinsic_param_mat = m_intrinsic_param_mat.clone();
245  m_new_intrinsic_param_mat.at<double>(0, 0) *= fx;
246  m_new_intrinsic_param_mat.at<double>(1, 1) *= fy;
247  //调节校正图中心，建议置于校正图中心
248  if (cx != 0 && cy != 0) {
249  m_new_intrinsic_param_mat.at<double>(0, 2) = cx;
250  m_new_intrinsic_param_mat.at<double>(1, 2) = cy;
251  }
252  cv::fisheye::undistortImage(src, dst, m_intrinsic_param_mat, m_distortion_coeffs, m_new_intrinsic_param_mat, cv::Size());
253  }

257  void remap(cv::cuda::GpuMat src, cv::cuda::GpuMat& dst, double fx = 1.0, double fy = 1.0, double cx = 0, double cy = 0)
258  {
259  //调节视场大小, 乘的系数越小视场越大
260  m_new_intrinsic_param_mat = m_intrinsic_param_mat.clone();
261  m_new_intrinsic_param_mat.at<double>(0, 0) *= fx;
262  m_new_intrinsic_param_mat.at<double>(1, 1) *= fy;
263  //调节校正图中心，建议置于校正图中心
264  if (cx != 0 && cy != 0) {
265  m_new_intrinsic_param_mat.at<double>(0, 2) = cx;
266  m_new_intrinsic_param_mat.at<double>(1, 2) = cy;
267  }
268  cv::cuda::GpuMat map1, map2;
269  cuda::cuda_init_undistort_rectify_map(m_intrinsic_param_mat_g, m_distortion_coeffs_g,
270  m_new_intrinsic_param_mat, src.size(), map1, map2);
271  cv::cuda::remap(src, dst, map1, map2, cv::INTER_LINEAR, cv::BORDER_CONSTANT);
272  }

302  double print_calibrate_errors() {
303  double total_err = 0.0;
304  std::vector<cv::Point2f> new_corners;
305  std::stringstream ss;
306  ss << "------------------------------\n" << "Average error of each calibration image(pix):\n";
307  for (int i = 0; i < m_chessboards.size(); i++)
308  {
309  //通过得到的摄像机内外参数,对空间的三维点进行重投影,得到新的投影点
310  cv::fisheye::projectPoints(m_object_points_set[i], new_corners, m_rotation_vectors[i], m_translation_vectors[i], m_new_intrinsic_param_mat, m_distortion_coeffs);
311  //计算新的投影点和旧的投影点之间的误差
312  double err = cv::norm(new_corners, m_corners_set[i], cv::NormTypes::NORM_L2);
313  total_err += err /= m_board_size.width*m_board_size.height;
314  ss << i << " : " << err << "\n";
315  }
316  double average_total_err = total_err / m_chessboards.size();
317  ss << "Total average error(pix): " << average_total_err << "\n"
318  << "------------------------------\n";
319  std::cout << ss.str();
320  LOGI(ss.str());
321  return average_total_err;
322  }
