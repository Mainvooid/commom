227  template <typename R, typename... Args>
228  std::function<R(Args...)> cache_fn(R(*func)(Args...))
229  {
230  auto result_map = std::make_shared<std::map<std::tuple<Args...>, R>>();
231  return ([=](Args... args) {//延迟执行
232  std::tuple<Args...> _args(args...);
233  if (result_map->find(_args) == result_map->end()) {
234  (*result_map)[_args] = func(args...);//未找到相同入参，执行函数刷新缓存
235  }
236  return (*result_map)[_args];//返回缓存
237  });
238  }
240  }// namespace detail

245  template <typename R, typename... Args>
246  std::function<R(Args...)> cache_fn(R(*func)(Args...), bool flush = false)
247  {
248  using function_type = std::function<R(Args...)>;
249  static std::unordered_map<decltype(func), function_type> functor_map;
250  if (flush) {//明确要求刷新缓存
251  return functor_map[func] = detail::cache_fn(func);
252  }
253  if (functor_map.find(func) == functor_map.end()) {
254  functor_map[func] = detail::cache_fn(func);//未找到相同函数，执行函数刷新缓存
255  }
256  return functor_map[func];//返回缓存
257  }

291  template<typename T>
292  auto fillDir(const T* dir, const T* separator = tvalue<T>("\\", L"\\"))
293  {
294  std::basic_string<T, std::char_traits<T>, std::allocator<T>> _dir = dir;
295  std::vector<const T*> separators = { tvalue<T>("\\",L"\\"), tvalue<T>("/",L"/") };
296  if (*separator == *separators[0]) {
297  separators.erase(separators.begin());
298  }
299  size_t n = 0;
300  while (true) {
301  n = _dir.find_first_of(separators[0]);
302  if (n == static_cast<size_t>(-1)) { break; }
303  _dir.replace(n, 1, separator);
304  }
305 
306  n = _dir.find_last_of(separator);
307  if (n == static_cast<size_t>(-1) || n != _dir.size() - 1) { _dir += separator; }//无结尾分隔符
308  return _dir;
309  }
