217  template <typename R, typename... Args>
218  std::function<R(Args...)> cache_fn(R(*func)(Args...))
219  {
220  auto result_map = std::make_shared<std::map<std::tuple<Args...>, R>>();
221  return ([=](Args... args) {//延迟执行
222  std::tuple<Args...> _args(args...);
223  if (result_map->find(_args) == result_map->end()) {
224  (*result_map)[_args] = func(args...);//未找到相同入参，执行函数刷新缓存
225  }
226  return (*result_map)[_args];//返回缓存
227  });
228  }
230  }// namespace detail

235  template <typename R, typename... Args>
236  std::function<R(Args...)> cache_fn(R(*func)(Args...), bool flush = false)
237  {
238  using function_type = std::function<R(Args...)>;
239  static std::unordered_map<decltype(func), function_type> functor_map;
240  if (flush) {//明确要求刷新缓存
241  return functor_map[func] = detail::cache_fn(func);
242  }
243  if (functor_map.find(func) == functor_map.end()) {
244  functor_map[func] = detail::cache_fn(func);//未找到相同函数，执行函数刷新缓存
245  }
246  return functor_map[func];//返回缓存
247  }

279  template<typename T>
280  auto fillDir(const T* dir, const T* separator = tvalue<T>("\\", L"\\"))
281  {
282  std::basic_string<T, std::char_traits<T>, std::allocator<T>> _dir = dir;
283  std::vector<const T*> separators = { tvalue<T>("\\",L"\\"), tvalue<T>("/",L"/") };
284  if (*separator == *separators[0]) {
285  separators.erase(separators.begin());
286  }
287  size_t n = 0;
288  while (true) {
289  n = _dir.find_first_of(separators[0]);
290  if (n == static_cast<size_t>(-1)) { break; }
291  _dir.replace(n, 1, separator);
292  }
293 
294  n = _dir.find_last_of(separator);
295  if (n == static_cast<size_t>(-1) || n != _dir.size() - 1) { _dir += separator; }//无结尾分隔符
296  return _dir;
297  }
