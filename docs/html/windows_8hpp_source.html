63  template<typename T>
64  static auto getWorkDir() noexcept
65  {
66  T current_exe_fname[MAX_PATH];
67  T _Dir[MAX_PATH];
68  T _Driver[sizeof(T) * 2];
69  T work_dir[MAX_PATH];
70  ::GetModuleFileName(nullptr, current_exe_fname, MAX_PATH);
71 #if defined(_UNICODE) or defined(UNICODE)
72  ::_wsplitpath_s(current_exe_fname, _Driver, sizeof(T) * 2, _Dir, MAX_PATH, nullptr, 0, nullptr, 0);
73 #else
74  ::_splitpath_s(current_exe_fname, _Driver, sizeof(T) * 2, _Dir, MAX_PATH, nullptr, 0, nullptr, 0);
75 #endif
76  ::wsprintf(work_dir, _T("%s%s"), _Driver, _Dir);
77  return std::basic_string<T, std::char_traits<T>, std::allocator<T>>(work_dir);
78  }

218  &featureLevel, //输出当前应用D3D特性等级

314  static HRESULT texture2d_to_texture2d(ID3D11Texture2D* pSrcTexture2D, ID3D11Texture2D** ppDstTexture2D, ID3D11Device* pDstDevice, HANDLE* dst_pHandle = nullptr)
315  {
316  D3D11_TEXTURE2D_DESC desc;
317  pSrcTexture2D->GetDesc(&desc);
318 
319  ComPtr<ID3D11Device> p_src_device;
320  pSrcTexture2D->GetDevice(p_src_device.GetAddressOf());
321  HRESULT hr = S_OK;
322  ComPtr <ID3D11Texture2D> p_new_src_texture;
323  if ((desc.MiscFlags & D3D11_RESOURCE_MISC_SHARED) != D3D11_RESOURCE_MISC_SHARED) {
324  ComPtr <ID3D11DeviceContext> p_ctx;
325  p_src_device->GetImmediateContext(&p_ctx);
326  desc.MiscFlags |= D3D11_RESOURCE_MISC_SHARED;
327  hr = p_src_device->CreateTexture2D(&desc, nullptr, p_new_src_texture.GetAddressOf());
328  if (FAILED(hr)) { return hr; }
329  p_ctx->CopyResource(p_new_src_texture.Get(), pSrcTexture2D);
330  p_ctx->Flush();
331  }
332  else {
333  p_new_src_texture = pSrcTexture2D;
334  }
335  ComPtr<IDXGIResource> pSharedResource;
336  hr = p_new_src_texture.As(&pSharedResource);//QueryInterface
337  if (FAILED(hr)) { return hr; }
338 
339  HANDLE handle;
340  hr = pSharedResource->GetSharedHandle(&handle);
341  if (FAILED(hr)) { return hr; }
342 
343  hr = pDstDevice->OpenSharedResource(handle, __uuidof(IDXGIResource), (void**)(pSharedResource.ReleaseAndGetAddressOf()));
344  if (FAILED(hr)) { return hr; }
345 
346  Release_s(*ppDstTexture2D);
347  hr = pSharedResource->QueryInterface(__uuidof(ID3D11Texture2D), reinterpret_cast<void**>(ppDstTexture2D));
348  if (FAILED(hr)) { return hr; }
349 
350  if (dst_pHandle != nullptr) { *dst_pHandle = handle; }
351  return S_OK;
352  }
